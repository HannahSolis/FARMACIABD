CREATE TABLE titulos(id_titulo number(8) NOT NULL PRIMARY KEY, titulacion varchar(50));

CREATE TABLE especialidad(id_especialidad number(8) NOT NULL PRIMARY KEY, especialidad varchar(50));

CREATE TABLE alergias_alimentos(id_alergias_alimenticias number(8) NOT NULL PRIMARY KEY, alimento varchar(50));

CREATE TABLE alergias_medicamentos(id_alergias_medicamentos number(8) NOT NULL PRIMARY KEY, medicamentos varchar(50));

CREATE TABLE tipo_sangre(id_tipo_sangre number(8) NOT NULL PRIMARY KEY, tipo varchar(5));

CREATE TABLE servicio(id_servicio number(8) NOT NULL PRIMARY KEY, Costo number(10), horario varchar(11), nombre varchar(50));

CREATE TABLE distribuidor(id_distribuidor number(8) NOT NULL PRIMARY KEY, nombre varchar(50), telefono number(10), email varchar(50));

CREATE TABLE tipo_medicamento(id_tipo_medicamento number(8) NOT NULL PRIMARY KEY, nombre varchar(50));

CREATE TABLE color_medicamento(id_color number(8) NOT NULL PRIMARY KEY, nombre varchar(50));

CREATE TABLE ubicacion(id_ubicacion number(6) NOT NULL PRIMARY KEY, localizacion varchar(50));

CREATE TABLE turno(id_turno number(8) NOT NULL PRIMARY KEY, turno varchar(10));

CREATE TABLE estudios(id_estudios number(8) NOT NULL PRIMARY KEY, id_titulo number(8), id_especialidad number(8), grado_estudios varchar(10), FOREIGN KEY(id_titulo) REFERENCES titulos(id_titulo), FOREIGN KEY(id_especialidad) REFERENCES especialidad(id_especialidad));

CREATE TABLE antecedentes(id_antecendentes number(8) NOT NULL PRIMARY KEY, id_alergias_alimenticias number(8), id_alergias_medicamentos number(8), id_tipo_sangre number(8), agudeza_visual number(8), cirugias varchar(50), FOREIGN KEY(id_alergias_alimenticias) REFERENCES alergias_alimentos(id_alergias_alimenticias), FOREIGN KEY(id_alergias_medicamentos) REFERENCES alergias_medicamentos(id_alergias_medicamentos), FOREIGN KEY(id_tipo_sangre) REFERENCES tipo_sangre(id_tipo_sangre));

CREATE TABLE paciente(id_paciente number(8) NOT NULL PRIMARY KEY, id_antecendentes number(8), estatura number(8), nombre varchar(50), FOREIGN KEY(id_antecendentes) REFERENCES antecedentes(id_antecendentes));

CREATE TABLE personal(id_personal number(8) NOT NULL PRIMARY KEY, nombre varchar(20), apellido_p varchar(20), apellido_m varchar(20), domicilio varchar(50), id_estudios number(8), id_turno number(8), edad number(2), telefono number(10), email varchar(50), FOREIGN KEY(id_estudios) REFERENCES estudios(id_estudios), FOREIGN KEY(id_turno) REFERENCES turno(id_turno));

CREATE TABLE cita(id_cita number(8) NOT NULL PRIMARY KEY, id_paciente number(8), peso number(8), id_servicio number(8), id_personal number(8), id_turno number(8), temperatura number(4), FOREIGN KEY(id_servicio) REFERENCES servicio(id_servicio), FOREIGN KEY(id_personal) REFERENCES personal(id_personal), FOREIGN KEY(id_turno) REFERENCES turno(id_turno));

CREATE TABLE presentacion(id_presentacion number(8) NOT NULL PRIMARY KEY, nombre varchar(50), gramaje number(8), id_color number(8), FOREIGN KEY(id_color) REFERENCES color_medicamento(id_color));

CREATE TABLE medicamentos(id_medicamento number(8) NOT NULL PRIMARY KEY, nombre varchar(50), id_tipo_medicamento number(8), precio number(10), contenido number(6), id_presentacion number(8), existencias number(10), caducidad varchar(10), id_ubicacion number(8), FOREIGN KEY(id_tipo_medicamento) REFERENCES tipo_medicamento(id_tipo_medicamento), FOREIGN KEY(id_presentacion) REFERENCES presentacion(id_presentacion), FOREIGN KEY(id_ubicacion) REFERENCES ubicacion(id_ubicacion));

CREATE TABLE orden_medicamento(id_orden number(8) NOT NULL PRIMARY KEY, cantidad number(4), id_medicamento number(8), FOREIGN KEY(id_medicamento) REFERENCES medicamentos(id_medicamento));

CREATE TABLE pedido(id_pedido number(8) NOT NULL PRIMARY KEY, fecha varchar(10), id_distribuidor number(8), id_personal number(8), id_orden number(8), FOREIGN KEY(id_orden) REFERENCES orden_medicamento(id_orden), FOREIGN KEY(id_distribuidor) REFERENCES distribuidor(id_distribuidor), FOREIGN KEY(id_personal) REFERENCES personal(id_personal));

CREATE TABLE prescripcion(id_prescripcion number(8) NOT NULL PRIMARY KEY, id_medicamento number(8), instrucciones varchar(50), FOREIGN KEY(id_medicamento) REFERENCES medicamentos(id_medicamento));

CREATE TABLE receta(id_receta number(8) NOT NULL PRIMARY KEY, id_cita number(8), id_personal number(8), id_prescripcion number(8), fecha varchar(10), FOREIGN KEY(id_cita) REFERENCES cita(id_cita), FOREIGN KEY(id_personal) REFERENCES personal(id_personal), FOREIGN KEY(id_prescripcion) REFERENCES prescripcion(id_prescripcion));

CREATE TABLE ventas(id_venta number(8) NOT NULL PRIMARY KEY, id_paciente number(8), fecha varchar(10), total number(10), FOREIGN KEY(id_paciente) REFERENCES paciente(id_paciente));

CREATE TABLE detalle_ventas(id_detalle_venta number(8) NOT NULL PRIMARY KEY, id_medicamento number(8), nombre varchar(50), precio number(8), cantidad number(8), costo number(8), id_venta number(8), FOREIGN KEY(id_medicamento) REFERENCES medicamentos(id_medicamento), FOREIGN KEY(id_venta) REFERENCES ventas(id_venta));
